from sealir.lam import LamBuilder

lambar = LamBuilder()

@lambar.lam_func
def func_body(x):
    a = lambar.expr("mul", x, lambar.expr("num", 2))
    b = lambar.expr("add", a, x)
    c = lambar.expr("sub", a, b)
    d = lambar.expr("mul", b, c)
    return d

print(lambar.format(func_body))
